> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 3个月内交付河道整治工程AI辅助设计MVP，包含地形解析、AI方案生成、CAD输出三大核心功能，以及支持项目记忆和规范检索的长期记忆系统。

```python
# backend/app/db/migrations/env.py
from logging.config import fileConfig
from sqlalchemy import engine_from_config, pool
from alembic import context
import sys
from pathlib import Path

sys.path.append(str(Path(__file__).parent.parent.parent))

from app.models.base import Base
from app.models import *  # 导入所有模型

config = context.config
if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata

def run_migrations_offline() -> None:
    url = config.get_main_option("sqlalchemy.url")
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
    )
    with context.begin_transaction():
        context.run_migrations()

def run_migrations_online() -> None:
    connectable = engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(connection=connection, target_metadata=target_metadata)
        with context.begin_transaction():
            context.run_migrations()

if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
```

```python
# backend/app/core/security.py
from pydantic import BaseModel
from typing import Set

class SecurityConfig(BaseModel):
    """安全配置"""
    # 文件上传
    allowed_extensions: Set[str] = {
        '.tif', '.tiff', '.shp', '.dxf',
        '.las', '.laz', '.xlsx', '.csv'
    }
    max_file_size: int = 500 * 1024 * 1024  # 500MB

    # AI Prompt
    max_input_length: int = 4000  # tokens
    injection_patterns: list = [
        '#', '---', 'instruction:', 'ignore:',
        'system:', 'you are:', 'act as:'
    ]

security_config = SecurityConfig()

def sanitize_user_input(text: str) -> str:
    """过滤Prompt注入常用模式"""
    for pattern in security_config.injection_patterns:
        text = text.replace(pattern, '')
    return text.strip()
```

```python
    '.shp': [b'\x00\x00\x27\x0a'],  # Shapefile
    '.dxf': [b'0', b'1', b'2', b'3', b'4', b'5', b'6', b'7', b'8', b'9'],  # DXF text
    '.xlsx': [b'PK\x03\x04'],  # ZIP-based
    '.csv': [b'\x00'],  # Text
}

def validate_file_extension(filename: str) -> Path:
    """验证文件扩展名"""
    path = Path(filename)
    suffix = path.suffix.lower()
    if suffix not in security_config.allowed_extensions:
        raise HTTPException(
            400,
            f"不支持的文件类型: {suffix}。允许的类型: {', '.join(security_config.allowed_extensions)}"
        )
    return path

def validate_file_size(content: bytes) -> None:
    """验证文件大小"""
    size = len(content)
    if size > security_config.max_file_size:
        raise HTTPException(
            413,
            f"文件超过大小限制: {size / 1024 / 1024:.1f}MB (最大 500MB)"
        )

def validate_file_magic(content: bytes, suffix: str) -> bool:
    """验证文件魔数"""
    if suffix not in MAGIC_BYTES:
        return True  # 没有定义的类型跳过检查

    valid_magics = MAGIC_BYTES[suffix]
    for magic_bytes in valid_magics:
        if content.startswith(magic_bytes):
            return True
    return False
```

```python
# backend/app/core/rule_engine.py
from typing import List, Optional, Dict, Any
from dataclasses import dataclass
from enum import Enum

class RuleCategory(str, Enum):
    FLOOD_STANDARD = "防洪标准"
    LEVEE_HEIGHT = "堤顶高程"
    SLOPE_RATIO = "坡比要求"
    CHANNEL_WIDTH = "河道底宽"
    REVETMENT = "护岸坡比"

@dataclass
class RuleViolation:
    rule_id: str
    rule_name: str
    expected: str
    actual: str
    suggestion: str

@dataclass
class ValidationResult:
    passed: bool
    violations: List[RuleViolation]

# MVP 核心规则定义
MVP_RULES = {
    RuleCategory.FLOOD_STANDARD: {
        "id": "flood_standard",
        "name": "防洪标准",
        "description": "根据工程等级确定设计洪水标准",
        "params": {
            "1级": 100,
            "2级": 50,
            "3级": 30,
            "4级": 20,
        },
        "unit": "重现期（年）"
    },
    RuleCategory.LEVEE_HEIGHT: {
        "id": "levee_height",
        "name": "堤顶高程",
        "description": "堤顶高程 = 设计洪水位 + 安全超高",
        "formula": "elevation = design_flood_level + freeboard",
        "params": {
            "1级": 1.0,
            "2级": 0.8,
            "3级": 0.6,
        },
        "unit": "m"
    },
    RuleCategory.SLOPE_RATIO: {
        "id": "slope_ratio",
        "name": "堤防坡比要求",
        "description": "根据堤高确定迎水面坡比",
        "params": {
            "粘土": {"<6m": "1:3.0", "6-10m": "1:3.5", ">10m": "1:4.0"},
            "壤土": {"<6m": "1:2.5", "6-10m": "1:3.0", ">10m": "1:3.5"},
        }
    },
    RuleCategory.CHANNEL_WIDTH: {
        "id": "channel_width",
        "name": "河道底宽",
        "description": "满足行洪能力的河道底宽要求",
        "params": {
            "最小底宽": 8.0,
        },
        "unit": "m"
    },
    RuleCategory.REVETMENT: {
        "id": "revetment_slope",
        "name": "护岸坡比",
        "description": "河道整治护岸坡比要求",
        "params": {
            "重力式护岸": "1:2.0",
            "坡式护岸": "1:2.5",
            "阶梯式护岸": "1:1.5",
        }
    }
}

class RuleEngine:
    def __init__(self):
        self.rules = MVP_RULES

    def validate_design(
        self,
        design_params: Dict[str, Any],
        engineering_grade: str = "2级"
    ) -> ValidationResult:
        """全面校验设计方案"""
        violations = []

        # 检查堤顶高程
        violations.extend(
            self._check_levee_height(design_params, engineering_grade)
        )

        # 检查坡比
        violations.extend(
            self._check_slope_ratio(design_params)
        )

        # 检查河道底宽
        violations.extend(
            self._check_channel_width(design_params)
        )

        return ValidationResult(
            passed=len(violations) == 0,
            violations=violations
        )

    def _check_levee_height(
        self,
        params: Dict[str, Any],
        grade: str
    ) -> List[RuleViolation]:
        """检查堤顶高程"""
        violations = []

        design_flood_level = params.get("design_flood_level")
        levee_top_elevation = params.get("levee_top_elevation")
        freeboard = self.rules[RuleCategory.LEVEE_HEIGHT]["params"].get(grade, 0.8)

        if design_flood_level and levee_top_elevation:
            expected_min = design_flood_level + freeboard
            if levee_top_elevation < expected_min:
                violations.append(RuleViolation(
                    rule_id="levee_height",
                    rule_name="堤顶高程校验",
                    expected=f"≥ {expected_min}m",
                    actual=f"{levee_top_elevation}m",
                    suggestion=f"堤顶高程应不低于 {expected_min}m（设计洪水位 {design_flood_level}m + 安全超高 {freeboard}m）"
                ))

        return violations

    def _check_slope_ratio(
        self,
        params: Dict[str, Any]
    ) -> List[RuleViolation]:
        """检查坡比要求"""
        violations = []

        slope_ratio = params.get("slope_ratio")
        levee_height = params.get("levee_height")

        if slope_ratio and levee_height:
            # 简化校验：实际坡比应小于等于规范要求
            required_slopes = self.rules[RuleCategory.SLOPE_RATIO]["params"]["壤土"]

            if levee_height < 6:
                required = required_slopes["<6m"]
            elif levee_height <= 10:
                required = required_slopes["6-10m"]
            else:
                required = required_slopes[">10m"]

            # 比较坡比（数值越大越缓）
            required_value = self._parse_slope_ratio(required)
            actual_value = self._parse_slope_ratio(slope_ratio)

            if actual_value < required_value:
                violations.append(RuleViolation(
                    rule_id="slope_ratio",
                    rule_name="坡比校验",
                    expected=f"≤ {required}",
                    actual=slope_ratio,
                    suggestion=f"当前堤高 {levee_height}m，坡比应不小于 {required}"
                ))

        return violations

    def _check_channel_width(
        self,
        params: Dict[str, Any]
    ) -> List[RuleViolation]:
        """检查河道底宽"""
        violations = []

        channel_width = params.get("channel_width")
        min_width = self.rules[RuleCategory.CHANNEL_WIDTH]["params"]["最小底宽"]

        if channel_width and channel_width < min_width:
            violations.append(RuleViolation(
                rule_id="channel_width",
                rule_name="河道底宽校验",
                expected=f"≥ {min_width}m",
                actual=f"{channel_width}m",
                suggestion=f"河道底宽不应小于 {min_width}m 以满足施工和检修要求"
            ))

        return violations

    def _parse_slope_ratio(self, slope_str: str) -> float:
        """解析坡比字符串，如 '1:3.0' -> 3.0"""
        if ":" in slope_str:
            return float(slope_str.split(":")[1])
        return float(slope_str)

    def get_applicable_rules(self, project_type: str) -> List[Dict]:
        """获取适用的规则列表"""
        return [
            {"id": rule["id"], "name": rule["name"], "description": rule["description"]}
            for rule in self.rules.values()
        ]
```

```python
        self.ai_service = ai_service

        self,
        project_id: uuid.UUID,
        terrain_id: uuid.UUID,
        regenerate: bool = False
    ) -> Design:
        """生成设计方案"""

        # 获取项目和地形数据
        project = self.db.query(Project).filter(Project.id == project_id).first()
        if not project:
            raise ProjectNotFound(str(project_id))

        terrain = self.db.query(Terrain).filter(Terrain.id == terrain_id).first()
        if not terrain:
            raise TerrainNotFound(str(terrain_id))

        # 获取适用规则
        rules = self.rule_engine.get_applicable_rules(project.type)

        # 检索相似历史方案
        similar_cases = self.memory_service.search_similar_designs(
            query=f"{project.type} {project.objectives.get('flood_standard', '')}",
            limit=3
        )

        # 调用 AI 生成
        design_json = self.ai_service.generate_design(
            terrain_features=terrain.features,
            project_objectives=project.objectives,
            rules_constraints=rules
        )

        # 规则引擎校验
        validation = self.rule_engine.validate_design(
            design_params=design_json.get("parameters", {}),
            engineering_grade=project.objectives.get("engineering_grade", "2级")
        )

        design_json["validation"] = {
            "passed": validation.passed,
            "violations": [
                {
                    "rule": v.rule_id,
                    "rule_name": v.rule_name,
                    "expected": v.expected,
                    "actual": v.actual,
                    "suggestion": v.suggestion
                }
                for v in validation.violations
            ]
        }

        # 保存设计方案
        design = Design(
            project_id=project_id,
            version=1,
            design_json=design_json,
            validation_report=validation.passed
        )
        self.db.add(design)
        self.db.commit()
        self.db.refresh(design)

        # 保存到记忆系统
        self.memory_service.save_design(project_id, design_json)

        return design

        return self.db.query(Design).filter(
            Design.project_id == project_id
```

```python
import os
import json
from pathlib import Path
from typing import Dict, List
from uuid import UUID
import ezdxf
from ezdxf.enums import TextEntityAlignment
from ezdxf.reltable import Linetype
from app.core.hydraulic_cad_standards import HYDRAULIC_LAYERS, DIM_STYLES
from app.db.database import SessionLocal
from app.models.design import Design

class CadService:
    def __init__(self):
        self.layers = HYDRAULIC_LAYERS
        self.dim_styles = DIM_STYLES

    def _register_linetypes(self, doc: ezdxf.document.Drawing) -> None:
        """注册水利工程标准线型"""
        linetypes = [
            ("CENTER", "中心线", [5.0, -2.5, 2.5, -2.5]),
            ("DASHED", "虚线", [5.0, -2.5]),
            ("PHANTOM", "双点划线", [12.5, -2.5, 2.5, -2.5, 2.5, -2.5]),
            ("DIVIDE", "点划线", [10.0, -2.5, 0, -2.5]),
        ]
        for name, desc, pattern in linetypes:
            if name not in doc.linetypes:
                doc.linetypes.add(name, desc, pattern)

    def _create_dxf_with_layers(self, path: Path) -> ezdxf.document.Drawing:
        """创建符合水利标准的DXF文档"""
        doc = ezdxf.new(dxfversion="AC2018")

        # 注册线型
        self._register_linetypes(doc)

        # 创建标准字体
        doc.styles.new("Standard", dxfattribs={"font": "romans.shx"})
        doc.styles.new("水利标注", dxfattribs={"font": "hztxt.shx", "width": 0.7})

        # 创建图层
        for layer_id, layer_info in self.layers.items():
            doc.layers.add(
                name=layer_id,
                dxfattribs={
                    "color": layer_info["color"],
                    "linetype": layer_info["linetype"],
                    "lineweight": layer_info.get("lineweight", 0.25) * 100  # DXF单位
                }
            )

        # 创建标注样式
        dim_style = doc.dimstyles.new("水利标注")
        dim_style.dimszf = self.dim_styles["水利标注"]["text_height"]
        dim_style.dimasz = self.dim_styles["水利标注"]["arrow_size"]
        dim_style.dimexo = self.dim_styles["水利标注"]["extension_line_offset"]

        return doc

        return doc

    def _generate_plan(
        self,
        layout: Dict,
        params: Dict,
        path: Path
    ):
        """生成总体布置图"""

        doc = self._create_dxf_with_layers(path)
        msp = doc.modelspace()

        # 绘制河道中心线（示例）
        treatment_length = layout.get("treatment_length", 5000)
        msp.add_line(
            (0, 0),
            (treatment_length, 0),
            dxfattribs={"layer": "RIVER_AXIS"}
        )

        # 绘制堤防线
        levee_offset = params.get("channel_width", 80) / 2
        msp.add_line(
            (0, levee_offset),
            (treatment_length, levee_offset),
            dxfattribs={"layer": "LEVEE_LINE"}
        )
        msp.add_line(
            (0, -levee_offset),
            (treatment_length, -levee_offset),
            dxfattribs={"layer": "LEVEE_LINE"}
        )

        # 添加标注
        msp.add_text(
            f"河道整治长度: {treatment_length}m",
            dxfattribs={"layer": "TEXT", "style": "Standard"}
        )

        doc.saveas(path)

    def _generate_cross_section(
        self,
        cross_sections: List[Dict],
        params: Dict,
        path: Path
    ):
        """生成横断面图"""

        doc = self._create_dxf_with_layers(path)
        msp = doc.modelspace()

        if not cross_sections:
            return

        cs = cross_sections[0]  # 取第一个断面
        levee_height = params.get("levee_height", 3.5)
        slope_ratio = params.get("slope_ratio", "1:3.0")
        slope_value = float(slope_ratio.split(":")[1])

        # 绘制地面线
        ground_elev = cs.get("ground_elevation", 45.5)
        msp.add_line(
            (-20, ground_elev),
            (20, ground_elev),
            dxfattribs={"layer": "RIVER_AXIS"}
        )

        # 绘制设计堤身
        levee_top = cs.get("design_elevation", 47.0)
        levee_width = levee_height * slope_value

        points = [
            (-levee_width, ground_elev),
            (-levee_width, levee_top),
            (levee_width, levee_top),
```

```python
        output_dir=output_dir
    )

    design.dxf_paths = drawings
    db.commit()

    return CadResponse(
        cad_id=design_id,
        drawings=drawings

    db: Session = Depends(get_db)
```

```python
        # 使用 pgvector 进行相似度搜索
        sql = text("""
            SELECT e.table_id, e.embedding,
                   1 - (e.embedding <=> :query_embedding) as similarity
            FROM embedding e
            WHERE e.table_type = 'design'
            ORDER BY e.embedding <=> :query_embedding
            LIMIT :limit
        """)

        result = self.db.execute(
            sql,
            {"query_embedding": str(query_embedding), "limit": limit}
        )

        similar_designs = []
        for row in result:
            design = self.db.query(Design).filter(Design.id == row.table_id).first()
            if design:
                similar_designs.append({
                    "design_id": str(design.id),
                    "similarity": row.similarity,

                })
```

```vue
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useRoute } from 'vue-router'
import { message } from 'ant-design-vue'

  UploadOutlined,
  DownloadOutlined,
  FilePdfOutlined,
  WarningOutlined

import { projectsApi } from '@/api/projects'
import { designsApi, type Design } from '@/api/designs'

const route = useRoute()
const projectId = route.params.id as string
const uploadRef = ref()

// 状态
const project = ref<any>(null)
const terrain = ref<any>(null)
const currentDesign = ref<Design | null>(null)
const generating = ref(false)
const validating = ref(false)
const generatingCad = ref(false)
const uploading = ref(false)
const uploadProgress = ref(0)
const cadDrawings = ref<Record<string, string> | null>(null)

// 超时配置
const GENERATE_TIMEOUT = 60000  // 60秒

const currentStep = computed(() => {
  if (!terrain.value) return 0
  if (!currentDesign.value) return 1
  if (!cadDrawings.value) return 2
  return 3
})

const validationPassed = computed(() =>
  currentDesign.value?.design_json?.validation?.passed ?? false
)

const validationStatus = computed(() =>
  validationPassed.value ? 'success' : 'error'
)

const violations = computed(() =>
  currentDesign.value?.design_json?.validation?.violations ?? []
)

const basisColumns = [
  { title: '规则', dataIndex: 'rule', key: 'rule' },
  { title: '规范依据', dataIndex: 'reference', key: 'reference' },
  { title: '校验结果', key: 'result' }
]

onMounted(async () => {
  await loadProject()
  await loadDesigns()
})

async function loadProject() {
  try {
    project.value = await projectsApi.get(projectId)
  } catch (error) {
    message.error('加载项目失败')
  }
}

async function loadDesigns() {
  try {
    const designs = await designsApi.list(projectId)
    if (designs.length > 0) {
      currentDesign.value = designs[0]
      cadDrawings.value = currentDesign.value.dxf_paths
    }
  } catch (error) {
    console.error('加载设计方案失败', error)
  }
}

async function handleTerrainUpload(file: File) {
  uploading.value = true
  uploadProgress.value = 0

  try {
    // 模拟进度
    const progressTimer = setInterval(() => {
      uploadProgress.value += 10
      if (uploadProgress.value >= 90) clearInterval(progressTimer)
    }, 200)

    const formData = new FormData()
    formData.append('file', file)

    // TODO: 替换为实际API调用
    // const response = await fetch(`/api/v1/projects/${projectId}/terrain`, {
    //   method: 'POST',
    //   body: formData
    // })
    // terrain.value = await response.json()

    // 模拟上传
    await new Promise(resolve => setTimeout(resolve, 2000))
    terrain.value = { id: 'mock', file_type: file.name.split('.').pop()?.toUpperCase(), features: { cross_sections_count: 25, elevation_range: [100, 150] } }
    message.success('地形文件上传成功')
  } catch (error) {
    message.error('文件上传失败，请重试')
  } finally {
    uploading.value = false
    uploadProgress.value = 0
  }

  return false  // 阻止默认上传
}

async function handleValidate() {
  if (!terrain.value) {
    message.warning('请先上传地形数据')
    return
  }

  validating.value = true

  try {
    // 带超时的生成
    const raceResult = await Promise.race([
      designsApi.generate(projectId, terrain.value.id),
      new Promise((_, reject) =>
        setTimeout(() => reject(new Error('timeout')), GENERATE_TIMEOUT)
      )
    ])

    currentDesign.value = raceResult
    message.success('AI校验完成')
  } catch (error: any) {
    if (error.message === 'timeout') {
      message.error('校验超时，请重试')
    } else {
      message.error('校验失败，请重试')
    }
  } finally {
    validating.value = false
  }
}

async function regenerateDesign() {
  generating.value = true
  try {
    currentDesign.value = await designsApi.generate(projectId, terrain.value.id, true)
    message.success('重新校验完成')
  } catch (error) {
    message.error('重新校验失败')
  } finally {
    generating.value = false
  }
}

async function handleGenerateCad() {
  if (!currentDesign.value) return

  generatingCad.value = true
  try {
    await designsApi.generateCad(currentDesign.value.id, ['plan', 'cross_section', 'profile'])
    cadDrawings.value = currentDesign.value.dxf_paths
    message.success('图纸生成成功')
  } catch (error) {
    message.error('图纸生成失败')
  } finally {
    generatingCad.value = false
  }
}

function focusUpload() {
  uploadRef.value?.focus()
}

function getViolationColor(severity: string) {
  const colors: Record<string, string> = {
    critical: 'red',
    high: 'orange',
    medium: 'gold',
    low: 'blue'
  }
  return colors[severity] || 'gold'
}

function getDrawingTypeName(type: string) {
  const names: Record<string, string> = {
    plan: '平面图',
    cross_section: '横断面图',
    profile: '纵断面图'
  }
  return names[type] || type
}
```
